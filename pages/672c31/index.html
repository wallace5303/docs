    main.go

    package main

import (
	"embed"

	"github.com/wallace5303/ee-go/eboot"

	"electron-egg/router"
)

var (
    // 嵌入静态资源
	//go:embed public/**
	staticFS embed.FS
)

func main() {
	// 初始化一个ego，之后你可以使用框架的各个模块
	ego := eboot.New(staticFS)

    // 开发者业务
	// 如：加载路由
	router.Api()

	// 最后运行
	ego.Run()
}

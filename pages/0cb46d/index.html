    # 使用方法

    const Log = require('ee-core/log');

# 高级语法，占位符
%s：字符串
%d：整数
%i：整数
%f：浮点数
%o：obj对象
%O：obj对象
%c：CSS样式

    • config [Object] - 日志配置

Log.coreLogger.error()
Log.coreLogger.debug()
Log.coreLogger.warn()
